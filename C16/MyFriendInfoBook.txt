import java.util.*;

class FriendInfoHandler
{
	private Friend[] fre;
	private int countFriend;
	String name;
	String phone;
	String address;
	String job;
	String major;
	
	FriendInfoHandler(int i)
	{
		countFriend = 0;
		fre = new Friend[i];
		
	}
	
	void addFriend(Friend fren)
	{
		fre[countFriend++] = fren;		
	}	
	
	void showAllBasic()
	{
		int i = 0;
		for(i = 0 ; i < countFriend ; i++)
		{
			System.out.println("hi");
			if(fre[i] == null)
			{
				System.out.println("it's null");
			}
			System.out.println("i is "+i);
			fre[i].basicInfo();
		}
	}
	void showAllInfo()
	{
		int i = 0;
		for(i = 0 ; i < countFriend ; i++)
		{
			System.out.println("hi");
			if(fre[i] == null)
			{
				System.out.println("it's null");
			}
			fre[i].showData();			
		}
	}
}

class Friend
{
	String name;
	String phone;
	String address;
	public Friend(String name, String phone, String address)
	{
		this.name = name;
		this.phone = phone;
		this.address = address;
	}
	public void showData()
	{
		System.out.println("이름 :"+name);
		System.out.println("번호 :"+phone);
		System.out.println("주소 :"+address);
	}	
	public void basicInfo()
	{
		System.out.println("이름 :"+name);
		System.out.println("번호 :"+phone);
	}
}

class HighFriend extends Friend
{
	String work;
	public HighFriend(String name, String phone, String address,String work)
	{
		super(name,phone,address);
		this.work = work;
	}
	public void showData()
	{
		super.showData();
		System.out.println("직업 :"+work);
	}
	public void basicInfo()
	{
		System.out.println("이름 :"+name);
		System.out.println("번호 :"+phone);
	}
}

class UnivFriend extends Friend
{
	String major;
	public UnivFriend(String name, String phone, String address,String major)
	{
		super(name,phone,address);
		this.major = major;
	}
	public void showData()
	{
		super.showData();
		System.out.println("전공 :"+major);
	}
	public void basicInfo()
	{
		System.out.println("이름 :"+name);
		System.out.println("번호 :"+phone);	
	}
}

class Friendinfo
{
	String name;
	String phone;
	String address;
	String major;
	String job;
	
	void inputInfo(int num)
	{
		Scanner sc = new Scanner(System.in);
		
		FriendInfoHandler freSet= new FriendInfoHandler(10);
		
		
		if(num == 1 || num == 2)
		{
			System.out.println("***********************");
			System.out.println("이름 :");
			name = sc.nextLine();
			System.out.println("번호 :");
			phone = sc.nextLine();
			System.out.println("주소 :");
			address = sc.nextLine();		
			if(num == 1)
			{
				System.out.println("직업 :");
				job = sc.nextLine();
							
				freSet.addFriend(new HighFriend(name,phone,address,job));
			}
			else if(num == 2)
			{
				System.out.println("전공 :");
				major = sc.nextLine();
				freSet.addFriend(new UnivFriend(name,phone,address,major));
			}
			
		}
		
		else if(num == 3)
		{
			freSet.showAllBasic();
		}
		else if(num == 4)
		{
			freSet.showAllInfo();
		}
			
	}
}
